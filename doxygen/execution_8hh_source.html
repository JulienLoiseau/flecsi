171  program_option(const char * section,
172  const char * flag,

174  std::initializer_list<initializer_value> values = {},
175  std::function<bool(const any &)> const & check = default_check) {
176 
177  auto semantic_ = boost::program_options::value(&value_);
178 
179  bool zero{false}, implicit{false};
180  for(auto const & v : values) {
181  if(v.first == option_default) {
182  semantic_->default_value(v.second);
183  }
184  else if(v.first == option_implicit) {
185  semantic_->implicit_value(v.second);
186  implicit = true;
187  }
188  else if(v.first == option_zero) {
189  semantic_->zero_tokens();
190  zero = true;
191  }
192  else if(v.first == option_multi) {

194  }
195  else {
196  flog_fatal("invalid value type");
197  } // if
198  } // for
