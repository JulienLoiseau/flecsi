175  template<class T>
176  std::size_t record(T & t) {
177  const auto tp = const_cast<void *>(static_cast<const void *>(&t));
178  if(auto & f = enumerated.front()) { // we have a free slot
179  auto & slot = *static_cast<void **>(f);
180  f = slot;
181  slot = tp;
182  return &slot - &f;
183  }
184  // NB: reallocation invalidates all zero of the free list pointers
185  enumerated.push_back(tp);
186  return enumerated.size() - 1;
187  }
189  void forget(std::size_t i) {
190  void *&f = enumerated.front(), *&p = enumerated[i];
191  p = f;
192  f = &p;
193  }
196  template<class T>
197  T & recall(std::size_t i) {
