168  template<class T>
169  std::size_t record(T & t) {
170  const auto tp = const_cast<void *>(static_cast<const void *>(&t));
171  if(auto & f = enumerated.front()) { // we have a free slot
172  auto & slot = *static_cast<void **>(f);
173  f = slot;
174  slot = tp;
175  return &slot - &f;
176  }
177  // NB: reallocation invalidates all zero of the free list pointers
178  enumerated.push_back(tp);
179  return enumerated.size() - 1;
180  }
182  void forget(std::size_t i) {
183  void *&f = enumerated.front(), *&p = enumerated[i];
184  p = f;
185  f = &p;
186  }
189  template<class T>
