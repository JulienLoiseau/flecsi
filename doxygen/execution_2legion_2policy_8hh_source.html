67 // Serialize a tuple of converted arguments (or references to existing
68 // arguments where possible). Note that is_constructible_v<const
69 // float&,const double&> is true, so we have to check
70 // is_constructible_v<float&,double&> instead.
71 template<class... PP, class... AA>
72 auto
73 serial_arguments(std::tuple<PP...> * /* to deduce PP */, AA &&... aa) {
74  static_assert((std::is_const_v<std::remove_reference_t<const PP>> && ...),
75  "Tasks cannot accept non-const references");
76  return utils::serial_put(std::tuple<std::conditional_t<
77  std::is_constructible_v<nonconst_ref_t<PP> &, nonconst_ref_t<AA>>,
