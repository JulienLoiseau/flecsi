49

52

100 entity_neighbors(const definition<D> & md, U && indices) {
101  flog_assert(from_dim == to_dim, "from_dim does not equal to to_dim");
102 
103  // Closure should include the initial set
104  std::set<size_t> closure(
105  std::forward<U>(indices).begin(), std::forward<U>(indices).end());
106 
107  using entityid = size_t;
108  using vertexid = size_t;
109 
110  // essentially vertex to cell and cell to cell through vertices
111  // connectivity.
112  std::map<vertexid, std::vector<entityid>> vertex2entities;
113  std::map<entityid, std::vector<entityid>> entity_neighbors;
114 
115  // build global entity to entity through # of shared vertices connectivity
116  for(size_t cell(0); cell < md.num_entities(from_dim); ++cell) {
117  std::map<entityid, size_t> cell_counts;
118  for(auto vertex : md.entities(from_dim, 0, cell)) {
119  // build vertex to cell connectivity, O(n_cells * n_polygon_sides)
120  // of insert().
121  vertex2entities[vertex].push_back(cell);
122 
123  // Count the number of times this cell shares a common vertex with
124  // some other cell. O(n_cells * n_polygon_sides * vertex to cells degrees)
125  for(auto other : vertex2entities[vertex]) {
126  // for (auto other : md.entities(0, from_dim, vertex)) {
127  if(other != cell)
128  cell_counts[other] += 1;
129  }
130  }
131 
132  for(auto count : cell_counts) {
133  if(count.second > thru_dim) {
134  // append cell to cell through "dimension" connectivity, we need to
135  // add both directions
136  entity_neighbors[cell].push_back(count.first);
137  entity_neighbors[count.first].push_back(cell);
138  }
139  }
140  }
141 
142  for(auto i : indices) {
143  for(auto n : entity_neighbors[i]) {
144  closure.insert(n);
145  }
146  }
