241  virtual void slice_task(const Legion::Mapping::MapperContext ctx,
242  const Legion::Task & task,
243  const Legion::Mapping::Mapper::SliceTaskInput & input,
244  Legion::Mapping::Mapper::SliceTaskOutput & output) {
245 
246  if(task.tag == FLECSI_MAPPER_SUBRANK_LAUNCH) {
247  // expect a 1-D index domain
248  assert(input.domain.get_dim() == 1);
249  // send the whole domain to our local processor
250  output.slices.resize(1);
251  output.slices[0].domain = input.domain;
252  output.slices[0].proc = task.target_proc;
253  return;
254  } // end if FLECSI_MAPPER_SUBRANK_LAUNCH
255 
256  if(task.tag == FLECSI_MAPPER_FORCE_RANK_MATCH) {
257  // expect a 1-D index domain - each point goes to the corresponding node
258  assert(input.domain.get_dim() == 1);
259  LegionRuntime::Arrays::Rect<1> r = input.domain.get_rect<1>();
260 
261  // go through all the CPU processors and find a representative for each
262  // node (i.e. address space)
263  std::map<int, Legion::Processor> targets;
264 
265  Legion::Machine::ProcessorQuery pq =
266  Legion::Machine::ProcessorQuery(machine).only_kind(
267  Legion::Processor::LOC_PROC);
268  for(Legion::Machine::ProcessorQuery::iterator it = pq.begin();
269  it != pq.end();
270  ++it) {
271  Legion::Processor p = *it;
272  int a = p.address_space();
273  if(targets.count(a) == 0)
274  targets[a] = p;
275  }
276 
277  output.slices.resize(1);
278  for(int a = r.lo[0]; a <= r.hi[0]; a++) {
279  assert(targets.count(a) > 0);
280  output.slices[0].domain = // Legion::Domain::from_rect<1>(
281  Legion::Rect<1>(a, a);
282  output.slices[0].proc = targets[a];
283  }
284  return;
285  } // FLECSI_MAPPER_FORCE_RANK_MATCH
286 
287  DefaultMapper::slice_task(ctx, task, input, output);
288  // end else
289  }
