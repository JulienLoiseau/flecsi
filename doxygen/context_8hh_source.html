158  inline int initialize_generic(int argc, char ** argv, bool dependent) {
159 
160  initialize_dependent_ = dependent;
161 
162  // Save command-line arguments
163  for(auto i(0); i < argc; ++i) {
164  argv_.push_back(argv[i]);
165  } // for
166 
167  std::string program(argv[0]);
168  program = "Basic Options (" + program.substr(program.rfind('/') + 1) + ")";
169 
170  boost::program_options::options_description master(program);
171  master.add_options()("help,h", "Print this message and exit.");
172 
173  // Add all of the descriptions to the main description
174  for(auto & od : descriptions_map_) {
175  master.add(od.second);
176  } // for
177 
178  boost::program_options::parsed_options parsed =
179  boost::program_options::command_line_parser(argc, argv)
180  .options(master)
181  .allow_unregistered()
182  .run();
183 
184  try {
185  boost::program_options::store(parsed, variables_map_);
186 
187  if(variables_map_.count("help")) {
188  if(process_ == 0) {
189  std::cout << master << std::endl;
190  } // if
191 
192  return status::help;
193  } // if
194 
195  boost::program_options::notify(variables_map_);
196  }
197  catch(boost::program_options::error & e) {
198  std::cerr << FLOG_COLOR_LTRED << "ERROR: " << FLOG_COLOR_RED << e.what()
199  << "!!!" << FLOG_COLOR_PLAIN << std::endl
200  << std::endl;
201  std::cerr << master << std::endl;
202  return status::command_line_error;
203  } // try
204 
205  unrecognized_options_ = boost::program_options::collect_unrecognized(
206  parsed.options, boost::program_options::include_positional);
207 
208  program_options_initialized_ = true;
209 
210 #if defined(FLECSI_ENABLE_FLOG)
211  if(flog_tags_ == "0") {
212  if(process_ == 0) {
213  std::cout << "Available tags (FLOG):" << std::endl;
214 
215  for(auto t : flog_tag_map()) {
216  std::cout << " " << t.first << std::endl;
217  } // for
218  } // if
219 
220  return status::help;
221  } // if
222 
223  if(flog_initialize(flog_tags_, flog_verbose_, flog_output_process_)) {
224  return status::error;
225  } // if
226 #endif
227 
228 #if defined(FLECSI_ENABLE_KOKKOS)
229  if(initialize_dependent_) {
230  // Need to capture status from this
231  Kokkos::initialize(argc, argv);
232  } // if
233 #endif
234 
235  return status::success;
236  } // initialize_generic
237
