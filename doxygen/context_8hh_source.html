207  auto const & master,

210  std::cout << "Usage: " << program << " ";
211 
212  size_t positional_count = positional_desc_.max_total_count();
213  size_t max_label_chars = std::numeric_limits<size_t>::min();
214 
215  for(size_t i{0}; i < positional_count; ++i) {
216  std::cout << "<" << positional_desc_.name_for_position(i) << "> ";
217 
218  const size_t size = positional_desc_.name_for_position(i).size();
219  max_label_chars = size > max_label_chars ? size : max_label_chars;
220  } // for
221 
222  max_label_chars += 2;
223 
224  std::cout << std::endl << std::endl;
225 
226  if(positional_count) {
227  std::cout << "Positional Options:" << std::endl;
228 
229  for(size_t i{0}; i < positional_desc_.max_total_count(); ++i) {
230  auto const & name = positional_desc_.name_for_position(i);
231  auto help = positional_help_.at(name);
232  std::cout << " " << name << " ";
233  std::cout << std::string(max_label_chars - name.size() - 2, ' ');
234

241
