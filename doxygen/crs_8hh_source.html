62  void erase(const std::vector<size_t> & ids) {
63 
64  if(ids.empty())
65  return;
66 
67  // assume sorted
68  assert(std::is_sorted(ids.begin(), ids.end()) &&
69  "entries to delete are not sorted");
70 
71  auto num_remove = ids.size();
72  auto num_offsets = size();
73 
74  std::vector<size_t> new_offsets(offsets.size() - num_remove);
75  std::vector<size_t> new_indices;
76  new_indices.reserve(indices.size());
77 
78  new_offsets[0] = 0;
79  auto delete_it = ids.begin();
80 
81  for(size_t iold = 0, inew = 0; iold < num_offsets; ++iold) {
82 
83  // skip deleted items
84  if(delete_it != ids.end()) {
85  if(*delete_it == iold) {
86  delete_it++;
87  continue;
88  }
89  }
90 
91  // keep otherwise
92  auto start = offsets[iold];
93  auto end = offsets[iold + 1];
94  for(auto j = start; j < end; ++j) {
95  new_indices.push_back(indices[j]);
96  }
97  new_offsets[inew + 1] = new_indices.size();
98  inew++;
99  }
100 
101  assert(new_offsets.size() == offsets.size() - num_remove);
102 
103  // resize arrays
104  std::swap(offsets, new_offsets);
105  std::swap(indices, new_indices);
106  } // erase
