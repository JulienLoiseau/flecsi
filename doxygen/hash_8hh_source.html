328 template<typename T, typename U>
329 inline constexpr T
330 string_hash(U && str, const std::size_t n) {
331  if(n == 0)
332  return 0;
333 
334  // String-to-integer hash function, based on prime numbers.
335  // References:
336  // https://stackoverflow.com/questions/8317508/hash-function-for-a-string
337  // https://planetmath.org/goodhashtableprimes
338 
339  const T P = 3145739; // prime
340  const T Q = 6291469; // prime, a bit less than 2x the first
341 
342  T h = 37; // prime
343  for(std::size_t i = 0; i < n; ++i)
344  h = (h * P) ^ (str[i] * Q);
345  return h;
346 } // string_hash
