167  void insert(entity_t & ent) {
168  // Find parent closest of the entity
169  key_t node_key = ent.key();
170  node_t * parent = find_parent(node_key);
171  assert(parent != nullptr);
172 
173  // It is not a leaf, need to insert intermediate node
174  if(!parent->is_leaf()) {
175  // Create the missing node
176  int depth = parent->key().depth() + 1;
177  node_key.truncate(depth);
178  int bit = node_key.last_value();
179  parent->add_bit_child(bit);
180 
181  // Insert this node and reinsert
182  node_map_.emplace_back(node_key, node_key);
183  auto * new_parent = &(node_map_.find(node_key)->second);
184  new_parent->set_leaf(true);
185  new_parent->insert(ent.global_id());
186  }
187  else {
188  // Conflict with a children
189  if(parent->size() == (1 << dimension)) {
190  refine_(parent);
191  insert(ent);
192  }
193  else {
194  parent->insert(ent.global_id());
195  }
196  }
197  parent = find_parent(node_key);
198  }

225  void cofm(node_t * b = nullptr, bool local = false) {
226  if(b == nullptr)
227  b = root();
228  // Find the sub particles on which we want to work
229  std::vector<node_t *> working_branches;
230  std::stack<node_t *> stk_remaining;
231  int level = 5;
232  std::stack<node_t *> stk;
233  stk.push(b);
234  while(!stk.empty()) {
235  node_t * c = stk.top();
236  int cur_level = c->key().depth();
237  stk.pop();
238  if(c->is_leaf()) {
239  working_branches.push_back(c);
240  }
241  else {
242  if(cur_level == level) {
243  working_branches.push_back(c);
244  }
245  else {
246  stk_remaining.push(c);
247  for(int i = (1 << dimension) - 1; i >= 0; --i) {
248  if(!c->as_child(i))
249  continue;
250  auto next = child_(c, i);
251  assert(next != nullptr);
252  stk.push(next);
253  }
254  }
255  }
256  }
257 
258  // Work in parallel on the sub branches
259  const int nwork = working_branches.size();
260 
261 #ifdef _OPENMP
262 #pragma omp parallel for
263 #endif
264  for(int b = 0; b < nwork; ++b) {
265  // Find the leave in order in these sub branches
266  std::stack<node_t *> stk1;
267  std::stack<node_t *> stk2;
268  stk1.push(working_branches[b]);
269  while(!stk1.empty()) {
270  node_t * cur = stk1.top();
271  stk1.pop();
272  stk2.push(cur);
273  // Push children to stk1
274  if(!cur->is_leaf()) {
275  for(int i = 0; i < (1 << dimension); ++i) {
276  if(!cur->as_child(i))
277  continue;
278  node_t * next = child_(cur, i);
279  stk1.push(next);
280  }
281  }
282  }
283  // Finish the highest part of the tree in serial
284  while(!stk2.empty()) {
285  node_t * cur = stk2.top();
286  stk2.pop();
287  update_COM(cur, local);
288  }
289  }
290  // Finish the high part of the tree on one thread
291  while(!stk_remaining.empty()) {
292  node_t * cur = stk_remaining.top();
293  stk_remaining.pop();
294  update_COM(cur, local);
295  }
296  }

301  void update_COM(node_t * b, bool /*local_only*/ = false) {
302  // Starting branch
303  element_t mass = 0;
304  point_t bmax{}, bmin{};
305  point_t coordinates{};
306  uint64_t nchildren = 0;
307  int owner = b->owner();
308  for(size_t d = 0; d < dimension; ++d) {
309  bmax[d] = -DBL_MAX;
310  bmin[d] = DBL_MAX;
311  }
312  if(b->is_leaf()) {
313  // For local branches, compute the radius
314  for(auto child : *b) {
315  auto ent = get(child);
316  owner = ent.owner();
317  ++nchildren;
318  element_t childmass = ent.mass();
319  for(size_t d = 0; d < dimension; ++d) {
320  bmax[d] = std::max(bmax[d], ent.coordinates()[d] + ent.radius() / 2.);
321  bmin[d] = std::min(bmin[d], ent.coordinates()[d] - ent.radius() / 2.);
322  }
323  coordinates += childmass * ent.coordinates();
324  mass += childmass;
325  }
326  if(mass > element_t(0))
327  coordinates /= mass;
328  }
329  else {
330  for(int i = 0; i < (1 << dimension); ++i) {
331  auto node = child_(b, i);
332  if(node == nullptr)
333  continue;
334  nchildren += node->sub_entities();
335  mass += node->mass();
336  if(node->mass() > 0) {
337  for(size_t d = 0; d < dimension; ++d) {
338  bmax[d] = std::max(bmax[d], node->bmax()[d]);
339  bmin[d] = std::min(bmin[d], node->bmin()[d]);
340  }
341  }
342  coordinates += node->mass() * node->coordinates();
343  }
344  if(mass > element_t(0))
345  coordinates /= mass;
346  }
347  b->set_sub_entities(nchildren);
348  b->set_coordinates(coordinates);
349  b->set_mass(mass);
350  b->set_bmin(bmin);
351  b->set_bmax(bmax);
352  assert(nchildren != 0);
353  }

474  void refine_(node_t * b) {
475  key_t pid = b->key();
476  size_t depth = pid.depth() + 1;
477  // For every children
478  char bit_child = 0;
479  for(auto ent : *b) {
480  key_t k = get(ent).key();
481  k.truncate(depth);
482  bit_child |= 1 << k.last_value();
483  node_map_.emplace_back(k, k);
484  }
485  max_depth_ = std::max(max_depth_, depth);
486  for(auto ent : *b) {
487  insert(get(ent));
488  }
489  b->set_leaf(false);
490  b->clear();
491  b->set_bit_child(bit_child);
492  }
